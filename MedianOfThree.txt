package program1;import java.util.Arrays;import java.util.Random;import java.util.Scanner;class MedianOfThree{	public static int comparisons;	public static int swaps;  // swap method swaps arr[i] and arr[j]static void swap(int[] arr, int i, int j){ int temp = arr[i]; arr[i] = arr[j]; arr[j] = temp; swaps++;}// partition sets median as pivot and places pivot in correct position// with values less than pivot <= pivot < values greater than pivotstatic int partition(int[] array, int low, int high){	// we are taking the median value as pivot	// take average of two indexes	int mid = (low + high) / 2;	// fetch values from array	int lowValue = array[low];	int midValue = array[mid];	int highValue = array[high];	// find min and max of the values	int min = Math.min(lowValue, Math.min(midValue, highValue));	int max = Math.max(lowValue, Math.max(midValue, highValue));	int sum = lowValue + midValue + highValue;	int median = sum - min - max;	int pivot = median;	int pivotIndex;	if (lowValue == median) 	{		pivotIndex = low;	} 	else if (midValue == median) 	{		pivotIndex = mid;	} 	else 	{		pivotIndex = high;	}	 // search for median in array and swap it to the end	 int i;	 for (i = 0; i < high; i++)	 {	     if (array[i] == pivot)	     {	    	 break;	     }	 }	 swap(array, i, high);	 // partition algorithm with pivot as median value	 i = low;	 for (int j = low; j<= high-1; j++)	 {		 comparisons++;		 if (array[j] <= pivot)		 {			 swap(array, i, j);			 i++;		 }	 }	 swap(array, i, high); 	 return(i);	}	 	static void quickSort(int[] array, int low, int high)	{		 if (low < high)	 {      	     int q = partition(array, low, high);	     quickSort(array, low, q - 1);	     quickSort(array, q + 1, high);	 }	}		static void printArray(int[] array, int size)	{		System.out.println("comparisons: " + comparisons);		System.out.println("exchanges: " + swaps);		/**		for(int i = 0; i < size; i++)	    System.out.print(array[i] + " ");  		System.out.println();		**/	}		public static void main(String[] args)	{		Scanner sc = new Scanner(System.in);		Random random = new Random();		System.out.println("Median-of-Three QuickSort with ascending list.");		System.out.println("Generating array of length:");		int number = sc.nextInt();		int[] array = new int[number];		for (int i = 0; i < array.length; i++)		 {			 array[i] = i;			 //System.out.println(arr[i]);		 }		int n = array.length;		quickSort(array, 0, n - 1);		printArray(array, n); 	 	}}
